  def draw_circle(x,y, radius = 5, mode = nil)
    if mode == :robot
      @gc.rgb_fg_color = @robot_color
    else
      @gc.rgb_fg_color = @black
    end
    @drawing_area.draw_arc(@gc, true, x-radius, y-radius, 2*radius, 2*radius, 0, 360*64)
    @gc.rgb_fg_color = @black
  end
