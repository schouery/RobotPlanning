    def decrease_key(i, key)
      raise "New key is greater than current key" if key > @list[i][P]
      @list[i][P] = key
      p = (i-1)/2
      while(i > 0 && @list[p][P] > @list[i][P])
        @list[p], @list[i] = @list[i], @list[p]
        i = p
        p = (i-1)/2
      end
    end
