  def recursive_draw(node, drawer)
    if node.class == YNode
      node.draw(drawer)
      recursive_draw(node.left_child, drawer)
      recursive_draw(node.right_child, drawer)
    elsif node.class == XNode
      recursive_draw(node.left_child, drawer)
      recursive_draw(node.right_child, drawer)
    else
      node.draw(drawer)
    end 
  end
