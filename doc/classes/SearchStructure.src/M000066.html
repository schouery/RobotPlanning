  def update_list(segment)
    list = []
    cell = find(segment, :segment => true)
    list << cell.trapezoid
    last = cell.trapezoid.rightp.x >= segment.finish.x 
    cell = cell.next_neighbour(segment)
    while(!last && !cell.nil? && cell.trapezoid.leftp.x < segment.finish.x)
      list << cell.trapezoid
      cell = cell.next_neighbour(segment)
    end
    list
  end
