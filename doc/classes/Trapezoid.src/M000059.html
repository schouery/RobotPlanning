  def calculate_center
    p1 = Point[@leftp.x, (@top_left_corner.y + @bottom_left_corner.y)/2]
    p2 = Point[@rightp.x, (@top_right_corner.y + @bottom_right_corner.y)/2]
    s = Segment[p1,p2]
    p3 = Point[(@leftp.x + @rightp.x)/2, 0]
    @center = s.x_projection(p3)
  end
