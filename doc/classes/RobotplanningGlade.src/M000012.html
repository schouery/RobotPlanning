  def save(filename)
    File.open(filename, 'w') do |f|
      @segments.each do |s|
        f.puts "#{s.start.x} #{s.start.y} #{s.finish.x} #{s.finish.y}"
      end
    end
    @current_file = filename
    @glade['mainwindow'].title = PROG_NAME + " - " + File.basename(filename)
  end
